CREATE TABLE "SYSTEM"."EFI_TABELA" 
   ("SETOR" VARCHAR2(100 BYTE), 
	"TRABALHO" VARCHAR2(100 BYTE), 
	"PECA" VARCHAR2(100 BYTE), 
	"TEMPO" VARCHAR2(20 BYTE)
   );


INSERT INTO EFI_TEMPOS (SETOR_ID, TRABALHO_ID, PECA_ID, TEMPO)
SELECT s.SETOR_ID, t.TRABALHO_ID, p.PECA_ID, aux.TEMPO FROM EFI_TABELA aux
JOIN EFI_SETORES s ON s.SETOR_NOME = aux.SETOR
JOIN EFI_TRABALHOS t ON t.TRABALHO_NOME = aux.TRABALHO
JOIN EFI_PECAS p ON p.PECA_DESCRICAO = aux.PECA;
